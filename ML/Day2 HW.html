scores= pd.DataFrame()
def predictions_plots(model,model_name, features= True):
    for i in range(7):
        model.fit(X_train, y_train['current_'+str(i)])
        prediction = model.predict(X_test)

        number = len(scores)
        scores.loc[number,'model'] = model_name+str(i)
        scores.loc[number,'score'] = mean_absolute_error(prediction,y_test['current_'+str(i)])
#         print(scores)
        
        fig = plt.figure(figsize=(12, 15))
        ax = plt.subplot(7, 2, i+1)
        plt.plot(prediction, label = 'Predicton '+model_name+str(i), color='r')
        plt.plot(list(y_test['current_'+str(i)]),label = 'Real', color='g')
        plt.legend()
        plt.show()

    # -----------------
        if features:  
            try:
                imp = pd.Series(list(model.coef_), name='importances') 
            except AttributeError:
                imp = pd.Series(list(model.feature_importances_), name='importances') 
            cols = pd.Series(past_columns, name='columns')
            df = pd.DataFrame(cols)
            df['imp'] = imp
            print(df)
    print(scores)
